205  std::vector<std::vector<double>> reg_cell(n_cells(), {0.0, 0.0});
206  std::size_t count = 0;
207  for (auto &T : _cells)
208  {
209  double hT = T->diam();
210  VectorRd<dimension> xT = T->center_mass();
211 
212  reg_cell[count][0] = hT / pow(T->measure(), 1.0 / dimension);
213 
214  double rhoT = hT;
215  std::vector<Face<dimension> *> faces = T->get_faces();
216  for (auto &F : faces)
217  {
218  double hF = F->diam();
219  VectorRd<dimension> xF = F->center_mass();
220  VectorRd<dimension> nTF = F->normal(); // sign does not matter
221 
222  reg_cell[count][0] = std::max(reg_cell[count][0], hT / hF);
223 
224  rhoT = std::min(rhoT, std::abs((xT - xF).dot(nTF))); // If xT is not in T, is this really a good measure?
225  }
226  reg_cell[count][1] = hT / rhoT;
227  ++count; //could just use iterators
228  }

232  {
233  value[0] = std::max(value[0], reg_cell[iT][0]);
234  value[1] = std::max(value[1], reg_cell[iT][1]);
235  }
