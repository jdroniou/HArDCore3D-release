56  // Dimension of polynomial spaces -- only specialisations are relevant
57  template<typename GeometricSupport>
58  inline size_t const DimPoly(int m);
59 
61  template<>
62  inline const size_t DimPoly<Cell>(const int m)
63  {
64  return (m >= 0 ? (m * (m + 1) * (2 * m + 1) + 9 * m * (m + 1) + 12 * (m + 1)) / 12 : 0);
65  };
66 
68  template<>
69  inline const size_t DimPoly<Face>(const int m)
70  {
71  return (m >= 0 ? (m + 1) * (m + 2) / 2 : 0);
72  }
73 
75  template<>
76  inline const size_t DimPoly<Edge>(const int m)
77  {
78  return (m >= 0 ? m + 1 : 0);
79  }

299  template<typename ContinuousFunction>
300  UVector HybridCore::interpolate(const ContinuousFunction& f, int cell_deg, size_t face_deg, size_t doe) const {
301 
302  size_t cell_deg_pos = std::max(cell_deg, 0);
303  assert(cell_deg <= CellDegree() || face_deg <= FaceDegree() || cell_deg > -1);
304 
305  // Sizes of local interpolation
306  size_t n_cell_dofs = DimPoly<Cell>(cell_deg_pos);
307  size_t n_total_cell_dofs = m_mesh->n_cells() * n_cell_dofs;
308  size_t n_face_dofs = DimPoly<Face>(face_deg);
309  size_t n_total_face_dofs = m_mesh->n_faces() * n_face_dofs;
310  Eigen::VectorXd XTF = Eigen::VectorXd::Zero( n_total_cell_dofs + n_total_face_dofs );
311 
312  // Face projections
313  std::function<void(size_t, size_t)> construct_all_face_projections
314  = [&](size_t start, size_t end)->void
315  {
316  for (size_t iF = start; iF < end; iF++) {
317  Face F = *m_mesh->face(iF);
318 
319  // Compute the L2 projection on face
320  QuadratureRule quadF = generate_quadrature_rule(F, doe);
321  boost::multi_array<double, 2> phiF_quadF = evaluate_quad<Function>::compute(FaceBasis(iF), quadF);
322  Eigen::VectorXd UF = l2_projection<PolyFaceBasisType>(f, FaceBasis(iF), quadF, phiF_quadF);
323 
324  // Fill in the complete vector of DOFs
325  XTF.segment(n_total_cell_dofs + iF * n_face_dofs, n_face_dofs) = UF;
326  }
327  };
328  parallel_for(m_mesh->n_faces(), construct_all_face_projections, m_use_threads);
329 
330  // Cell projections
331  std::function<void(size_t, size_t)> construct_all_cell_projections
332  = [&](size_t start, size_t end)->void
333  {
334  for (size_t iT = start; iT < end; iT++) {
335  Cell* cell = m_mesh->cell(iT);
336 
337  // L2 projection in cell
338  // Mass matrix in cell
339  QuadratureRule quadT = generate_quadrature_rule(*cell, doe);
340  boost::multi_array<double, 2> phiT_quadT = evaluate_quad<Function>::compute(CellBasis(iT), quadT);
341  Eigen::MatrixXd MT = compute_gram_matrix(phiT_quadT, phiT_quadT, quadT, n_cell_dofs, n_cell_dofs, "sym");
342 
343  // Vector of integrals of f against basis functions
344  Eigen::VectorXd bT = Eigen::VectorXd::Zero(n_cell_dofs);
345  for (size_t i = 0; i < n_cell_dofs; i++) {
346  for (size_t iqn = 0; iqn < quadT.size(); iqn++){
347  bT(i) += quadT[iqn].w * phiT_quadT[i][iqn] * f(quadT[iqn].vector());
348  }
349  }
350 
351  // Vector of coefficients (on cell basis functions) of the L2(T) projection of f
352  Eigen::VectorXd UT = MT.ldlt().solve(bT);
353 
354  // Fill in the complete vector of DOFs
355  size_t offset_T = iT * n_cell_dofs;
356  XTF.segment(offset_T, n_cell_dofs) = UT;
357 
358  // Special case of L=-1, we replace the previously computed cell value with the proper average of face values
359  if ( cell_deg==-1 ){
360  size_t nfaces = cell->n_faces();
361  // Weights corresponding to the values in cells/on faces
362  Eigen::VectorXd barycoefT = compute_weights(iT);
363  // We transform these weights so that they apply to the coefficients on the basis functions
364  VectorRd xT = cell->center_mass();
365  double phiT_cst = CellBasis(iT).function(0, xT);
366  for (size_t ilF = 0; ilF < nfaces; ilF++){
367  VectorRd xF = cell->face(ilF)->center_mass();
368  size_t iF = cell->face(ilF)->global_index();
369  double phiF_cst = FaceBasis(iF).function(0, xF);
370  barycoefT(ilF) *= phiF_cst / phiT_cst;
371  }
372 
373  XTF(iT) = 0;
374  for (size_t ilF = 0; ilF < nfaces; ilF++) {
375  size_t iF = cell->face(ilF)->global_index();
376  XTF(iT) += barycoefT(ilF) * XTF(n_total_cell_dofs + iF);
377  }
378  }
379 
380  }
381  };
382  parallel_for(m_mesh->n_cells(), construct_all_cell_projections, m_use_threads);
383 
384  return UVector(XTF, *get_mesh(), cell_deg, face_deg);
385  }
