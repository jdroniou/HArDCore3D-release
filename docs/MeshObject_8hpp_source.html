178  template <size_t space_dim, size_t object_dim>
179  VectorRd<space_dim> simplex_center_mass(Simplex<space_dim, object_dim> simplex)
180  {
181  VectorRd<space_dim> center_mass = Eigen::VectorXd::Zero(space_dim);
182 
183  for (auto& coord : simplex)
184  {
185  for (size_t i = 0; i < space_dim; ++i)
186  {
187  center_mass(i) += coord(i);
188  }
189  }
190  return center_mass / (object_dim + 1);
191  }

193  template <size_t space_dim, size_t object_dim>
194  double simplex_measure(Simplex<space_dim, object_dim> simplex)
195  {
196  Eigen::MatrixXd CM_mat = Eigen::MatrixXd::Zero(object_dim + 2, object_dim + 2);
197  for (size_t i = 0; i < object_dim + 1; ++i)
198  {
199  VectorRd<space_dim> i_coords = simplex[i];
200  for (size_t j = i + 1; j < object_dim + 1; ++j)
201  {
202  VectorRd<space_dim> j_coords = simplex[j];
203  double norm = (j_coords - i_coords).norm();
204  CM_mat(i, j) = norm * norm;
205  CM_mat(j, i) = CM_mat(i, j);
206  }
207  CM_mat(i, object_dim + 1) = 1;
208  CM_mat(object_dim + 1, i) = 1;
209  }
210 
211  // Calculate Cayley-Menger determinant
212  double det = CM_mat.determinant();
213  double scaling = std::pow(Math::factorial(object_dim), 2) * std::pow(2, object_dim);
214 
215  return std::sqrt(std::abs(det / scaling));
216  }

219  MeshObject<space_dim, object_dim>::MeshObject(size_t index, Simplices<space_dim, object_dim> simplices)
220  : _index(index), _simplices(simplices)
221  {
222  assert(space_dim >= object_dim);
223  assert(space_dim >= 2);
224 
225  if (object_dim == 0 || object_dim == 1)
226  {
227  assert(simplices.size() == 1);
228  }
229 
230  _is_boundary = false;
231 
232  _measure = 0.0;
233  _center_mass = Eigen::VectorXd::Zero(space_dim);
234  std::vector<VectorRd<space_dim>> vertex_coords;
235  for (auto& simplex : simplices)
236  {
237  // assert(simplex.size() == _dim + 1);
238  double tmp = simplex_measure<space_dim, object_dim>(simplex);
239  _measure += tmp;
240  _center_mass += tmp * simplex_center_mass<space_dim, object_dim>(simplex);
241  for (auto& coord : simplex)
242  {
243  if (std::find(vertex_coords.begin(), vertex_coords.end(), coord) == vertex_coords.end())
244  {
245  vertex_coords.push_back(coord); // if coord not already in vertex_coords, add it
246  }
247  }
248  }
249  _center_mass /= _measure;
250 
251  _diameter = 0.0;
252  for (auto it = vertex_coords.begin(); it != vertex_coords.end(); ++it)
253  {
254  for (auto jt = it; jt != vertex_coords.end(); ++jt)
255  {
256  _diameter = std::max(_diameter, (*it - *jt).norm()); // probably more efficient methods
257  }
258  }
259 
260  if (object_dim == space_dim - 1) // find the normal
261  {
262  Simplex<space_dim, object_dim> simplex = _simplices[0];
263  Eigen::MatrixXd mat = Eigen::MatrixXd::Zero(object_dim, space_dim);
264  for (size_t i = 0; i < object_dim; ++i)
265  {
266  for (size_t j = 0; j < space_dim; ++j)
267  {
268  mat(i, j) = simplex[i + 1][j] - simplex[0][j]; // find d-1 linearly independent vectors in the face, and put them in each col of mat
269  }
270  }
271 
272  // all normal vectors lie in null space of mat
273  Eigen::FullPivLU<Eigen::MatrixXd> lu(mat);
274  _normal = lu.kernel().normalized(); // not most efficient routine
275  }
276  }

364  void MeshObject<space_dim, object_dim>::construct_face_normals() // not very efficient - probably room for improvement
365  {
366  assert(object_dim == space_dim);
367  for (size_t iF = 0; iF < _faces.size(); ++iF)
368  {
369  VectorRd<space_dim> normal = _faces[iF]->normal();
370  Simplex<space_dim, object_dim - 1> face_simplex = _faces[iF]->get_simplices()[0];
371  VectorRd<space_dim> center = Eigen::VectorXd::Zero(space_dim);
372  double count;
373  for (auto& cell_simplex : this->get_simplices())
374  {
375  count = 0;
376  for (size_t i = 0; (i < cell_simplex.size()) && count < 2; ++i)
377  {
378  if (std::find(face_simplex.begin(), face_simplex.end(), cell_simplex[i]) == face_simplex.end())
379  {
380  ++count;
381  }
382  }
383  if (count == 1) // only don't share one coordinate
384  {
385  center = simplex_center_mass<space_dim, object_dim>(cell_simplex);
386  break;
387  }
388  }
389  assert(count == 1);
390  // _face_directions.push_back(Math::sgn((_faces[iF]->center_mass() - _center_mass).dot(normal))); // star shaped wrt center mass
391  _face_directions.push_back(Math::sgn((_faces[iF]->center_mass() - center).dot(normal)));
392  assert(_face_directions[iF] != 0);
393  }
