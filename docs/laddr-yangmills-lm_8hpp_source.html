865  )

867  // Matrices and vectors
868  std::vector<Eigen::SparseMatrix<double>> systems;
869  systems.reserve(size_systems.size());
870  std::vector<Eigen::VectorXd> vectors;
871  for (auto size : size_vectors){
872  vectors.emplace_back(Eigen::VectorXd::Zero(size));
873  }

875  if (use_threads) {

877  unsigned nb_threads_hint = std::thread::hardware_concurrency();
878  unsigned nb_threads = nb_threads_hint == 0 ? 8 : (nb_threads_hint);
879 
880  // Compute the batch size and the remainder
881  unsigned batch_size = nb_elements / nb_threads;
882  unsigned batch_remainder = nb_elements % nb_threads;
883 
884  // Create vectors of triplets and vectors
885  std::vector<std::vector<std::list<Eigen::Triplet<double> > > > triplets(nb_threads + 1, std::vector<std::list<Eigen::Triplet<double> > >(size_systems.size()));
886  std::vector<std::vector<Eigen::VectorXd>> vecs(nb_threads + 1, vectors);
887 
888  // Assign a task to each thread
889  std::vector<std::thread> my_threads(nb_threads);
890  for (unsigned i = 0; i < nb_threads; ++i) {
891  int start = i * batch_size;
892  my_threads[i] = std::thread(batch_local_assembly, start, start + batch_size, &triplets[i], &vecs[i]);
893  }
894 
895  // Execute the elements left
896  int start = nb_threads * batch_size;
897  batch_local_assembly(start, start + batch_remainder, &triplets[nb_threads], &vecs[nb_threads]);
898 
899  // Wait for the other threads to finish their task
900  std::for_each(my_threads.begin(), my_threads.end(), std::mem_fn(&std::thread::join));
901

903  for (size_t i = 0; i < size_systems.size(); i++){

905  size_t n_triplets = 0;

907  n_triplets += triplets_thread[i].size();
908  }
909  std::vector<Eigen::Triplet<double> > all_triplets(n_triplets);
910  auto triplet_index = all_triplets.begin();

912  triplet_index = std::copy(triplets_thread[i].begin(), triplets_thread[i].end(), triplet_index);
913  }

915  }

918  for (auto vec_thread : vecs){

920  }
921  }
922
