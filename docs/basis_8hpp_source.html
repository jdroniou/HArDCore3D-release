81  struct MonomialPowers<Cell>
82  {
83  // Powers of homogeneous polynomials of degree L
84  static std::vector<VectorZd> homogeneous(const size_t L)
85  {
86  std::vector<VectorZd> powers;
87  powers.reserve( PolynomialSpaceDimension<Cell>::Poly(L) - PolynomialSpaceDimension<Cell>::Poly(L-1) );
88  for (size_t i = 0; i <= L; i++)
89  {
90  for (size_t j = 0; i + j <= L; j++)
91  {
92  powers.push_back(VectorZd(i, j, L - i - j));
93  } // for j
94  } // for i
95  return powers;
96  }
97 
98  // Powers for degrees from 0 to degree
99  static std::vector<VectorZd> complete(const size_t degree)
100  {
101  std::vector<VectorZd> powers;
102  powers.reserve( PolynomialSpaceDimension<Cell>::Poly(degree) );
103  for (size_t l = 0; l <= degree; l++)
104  {
105  std::vector<VectorZd> pow_hom = homogeneous(l);
106  for (VectorZd p : pow_hom){
107  powers.push_back(p);
108  }
109  } // for l
110  return powers;
111  }
112  };

564  static const bool hasGradient = ScalarFamilyType::hasGradient;
565  // We know how to compute the curl and divergence if gradient is available, and
566  // if we tensorize at the dimension of the space
567  static const bool hasDivergence = (ScalarFamilyType::hasGradient && N == dimspace);
568  static const bool hasCurl = (ScalarFamilyType::hasGradient && N == dimspace);
569 
570  TensorizedVectorFamily(const ScalarFamilyType &scalar_family)

581  }
582

585  {
586  static_assert(hasFunction, "Call to function() not available");
587 
588  FunctionValue ek = Eigen::Matrix<double, N, 1>::Zero();
589  ek(i / m_scalar_family.dimension()) = 1.;
590  return ek * m_scalar_family.function(i % m_scalar_family.dimension(), x);

595  {

597 
598  FunctionValue ek = Eigen::Matrix<double, N, 1>::Zero();
599  ek(i / m_scalar_family.dimension()) = 1.;
600  return ek * ancestor_value_quad[i % m_scalar_family.dimension()][iqn];
601  }

605  {
606  static_assert(hasGradient, "Call to gradient() not available");
607 
608  GradientValue G = Eigen::Matrix<double, N, dimspace>::Zero();

610  return G;
611  }

627

1699  template <typename T>
1700  Eigen::MatrixXd gram_schmidt(
1701  boost::multi_array<T, 2> &basis_eval,
1702  const std::function<double(size_t, size_t)> &inner_product
1703  )
1704  {
1705  auto induced_norm = [inner_product](size_t i) -> double {
1706  return std::sqrt(inner_product(i, i));
1707  };
1708 
1709  // Number of basis functions
1710  size_t Nb = basis_eval.shape()[0];
1711  // Number of quadrature nodes
1712  size_t Nqn = basis_eval.shape()[1];
1713 
1714  Eigen::MatrixXd B = Eigen::MatrixXd::Zero(Nb, Nb);
1715 
1716  // Normalise the first element
1717  double norm = induced_norm(0);
1718  for (size_t iqn = 0; iqn < Nqn; iqn++)
1719  {
1720  basis_eval[0][iqn] /= norm;
1721  } // for iqn
1722  B(0, 0) = 1. / norm;
1723 
1724  for (size_t ib = 1; ib < Nb; ib++)
1725  {
1726  // 'coeffs' represents the coefficients of the ib-th orthogonal function on the ON basis functions 0 to ib-1
1727  Eigen::RowVectorXd coeffs = Eigen::RowVectorXd::Zero(ib);
1728  for (size_t pb = 0; pb < ib; pb++)
1729  {
1730  coeffs(pb) = -inner_product(ib, pb);
1731  } // for pb
1732 
1733  // store the values of the orthogonalised ib-th basis function
1734  for (size_t pb = 0; pb < ib; pb++)
1735  {
1736  for (size_t iqn = 0; iqn < Nqn; iqn++)
1737  {
1738  basis_eval[ib][iqn] += coeffs(pb) * basis_eval[pb][iqn];
1739  } // for iqn
1740  } // for pb
1741 
1742  // normalise ib-th basis function
1743  double norm = induced_norm(ib);
1744  for (size_t iqn = 0; iqn < Nqn; iqn++)
1745  {
1746  basis_eval[ib][iqn] /= norm;
1747  } // for iqn
1748  coeffs /= norm;
1749  // Compute ib-th row of B.
1750  // B.topLeftCorner(ib, ib) contains the rows that represent the ON basis functions 0 to ib-1 on
1751  // the original basis. Multiplying on the left by coeffs gives the coefficients of the ON basis
1752  // functions on the original basis functions from 0 to ib-1.
1753  B.block(ib, 0, 1, ib) = coeffs * B.topLeftCorner(ib, ib);
1754  B(ib, ib) = 1. / norm;
1755  }
1756  return B;
1757  }
