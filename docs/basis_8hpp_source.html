1770  template <typename T>
1771  Eigen::MatrixXd gram_schmidt(
1772  boost::multi_array<T, 2> &basis_eval,
1773  const std::function<double(size_t, size_t)> &inner_product
1774  )
1775  {
1776  auto induced_norm = [inner_product](size_t i) -> double {
1777  return std::sqrt(inner_product(i, i));
1778  };
1779 
1780  // Number of basis functions
1781  size_t Nb = basis_eval.shape()[0];
1782  // Number of quadrature nodes
1783  size_t Nqn = basis_eval.shape()[1];
1784 
1785  Eigen::MatrixXd B = Eigen::MatrixXd::Zero(Nb, Nb);
1786 
1787  // Normalise the first element
1788  double norm = induced_norm(0);
1789  for (size_t iqn = 0; iqn < Nqn; iqn++)
1790  {
1791  basis_eval[0][iqn] /= norm;
1792  } // for iqn
1793  B(0, 0) = 1. / norm;
1794 
1795  for (size_t ib = 1; ib < Nb; ib++)
1796  {
1797  // 'coeffs' represents the coefficients of the ib-th orthogonal function on the ON basis functions 0 to ib-1
1798  Eigen::RowVectorXd coeffs = Eigen::RowVectorXd::Zero(ib);
1799  for (size_t pb = 0; pb < ib; pb++)
1800  {
1801  coeffs(pb) = -inner_product(ib, pb);
1802  } // for pb
1803 
1804  // store the values of the orthogonalised ib-th basis function
1805  for (size_t pb = 0; pb < ib; pb++)
1806  {
1807  for (size_t iqn = 0; iqn < Nqn; iqn++)
1808  {
1809  basis_eval[ib][iqn] += coeffs(pb) * basis_eval[pb][iqn];
1810  } // for iqn
1811  } // for pb
1812 
1813  // normalise ib-th basis function
1814  double norm = induced_norm(ib);
1815  for (size_t iqn = 0; iqn < Nqn; iqn++)
1816  {
1817  basis_eval[ib][iqn] /= norm;
1818  } // for iqn
1819  coeffs /= norm;
1820  // Compute ib-th row of B.
1821  // B.topLeftCorner(ib, ib) contains the rows that represent the ON basis functions 0 to ib-1 on
1822  // the original basis. Multiplying on the left by coeffs gives the coefficients of the ON basis
1823  // functions on the original basis functions from 0 to ib-1.
1824  B.block(ib, 0, 1, ib) = coeffs * B.topLeftCorner(ib, ib);
1825  B(ib, ib) = 1. / norm;
1826  }
1827  return B;
1828  }
