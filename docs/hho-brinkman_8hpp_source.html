75  double Cf(const Cell & T) const
76  {
77  double muT = mu.value(T, T.center_mass());
78  double kappainvT = kappainv.value(T, T.center_mass());
79 
80  // Thres is the level at which the parameter is put at 0; note that Cf is also bounded below by thres
81  // to avoid singularities when taking Cf^{-1}.
82  double thres = 1e-14;
83  return (muT > thres ? std::max(thres, kappainvT * std::pow(T.diam(), 2) / muT) : 1./thres);
84  }

86  ViscosityType mu;
87  PermeabilityInvType kappainv;
88  };

578  static Brinkman::VelocityGradientType
579  regimes_graduD = [](const VectorRd & x) -> MatrixRd {
580  MatrixRd M = MatrixRd::Zero();
581 
582  if (scaling_kappainv>eps){
583  M.row(0) << sin(2. * PI * x(0)) * sin(2. * PI * x(1)) * sin(2. * PI * x(2)),
584  -cos(2. * PI * x(0)) * cos(2. * PI * x(1)) * sin(2. * PI * x(2)),
585  -cos(2. * PI * x(0)) * sin(2. * PI * x(1)) * cos(2. * PI * x(2));
586  M.row(1) << -cos(2. * PI * x(0)) * cos(2. * PI * x(1)) * sin(2. * PI * x(2)),
587  sin(2. * PI * x(0)) * sin(2. * PI * x(1)) * sin(2. * PI * x(2)),
588  -sin(2. * PI * x(0)) * cos(2. * PI * x(1)) * cos(2. * PI * x(2));
589  M.row(2) << -cos(2. * PI * x(0)) * sin(2. * PI * x(1)) * cos(2. * PI * x(2)),
590  -sin(2. * PI * x(0)) * cos(2. * PI * x(1)) * cos(2. * PI * x(2)),
591  sin(2. * PI * x(0)) * sin(2. * PI * x(1)) * sin(2. * PI * x(2));
592  M *= 4. * std::pow(PI, 2) / scaling_kappainv;
593  }
